// Tests assignment operators (+=, -=, *=, /=, %=, ^=) and unary ++/-- on designators
main

int x;
int y;
int z;
int[3] arr;
bool b;

{
    x = 2;
    y = 5;
    z = 3;
    b = false;

    x += y;       // 7
    y *= z;       // 15
    y /= x;       // 2 (integer division)
    z ^= x;       // pow assign: z = z ^ x
    x %= y;       // modulo

    arr[0] = 1;
    arr[1] = 2;
    arr[2] = 3;

    arr[1] ++;    // 3
    arr[2] --;    // 2

    b = not b and true; // true
}.
